CORRECT CODE (WITHOUT NAMESPACE.STD)
#include <iostream>

int main()
{
    int p, t;
    float r, si;

    std::cout << "Enter the Principal amount: ";
    std::cin >> p;

    std::cout << "Enter the Rate of Interest: ";
    std::cin >> r;

    std::cout << "Enter the Time in years: ";
    std::cin >> t;

    si = (p * r * t) / 100;

    std::cout << "The Simple Interest is: " << si << std::endl;

    return 0;
}

#include <iostream>

int main()
{

    cout << "Enter the Time in years: ";
    cin >> t;

    si = (p * r * t) / 100;

    cout << "The Simple Interest is: " << si << endl;
